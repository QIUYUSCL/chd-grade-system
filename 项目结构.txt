web-frontend/（对应“Web前端网络设计”）
├── public/                # 静态资源（图标、打印模板）
├── src/
│   ├── api/               # Axios请求封装（按模块划分：user.js、score.js、course.js）
│   ├── assets/            # 样式、图片资源
│   ├── components/        # 公共组件（分页、打印组件、成绩卡片、XSS过滤组件）
│   ├── router/            # 路由配置（角色路由守卫，控制教师/学生/管理员菜单可见性）
│   ├── store/             # Pinia状态管理（用户信息、权限标识、全局配置）
│   ├── views/             # 页面视图（按角色划分）
│   │   ├── teacher/       # 教师端：成绩录入（含系数设置）、成绩分析表、往年成绩查询
│   │   ├── student/       # 学生端：成绩查看（正考/补考格式）、排名、挂科统计
│   │   ├── admin/         # 管理员端：成绩管理（小撤销/大撤销）、人员维护、权限分配、安全日志
│   │   └── login/         # 登录页面（JWT认证）
│   ├── utils/             # 工具类（加密、XSS过滤、日期格式化、打印工具）
│   ├── App.vue            # 根组件
│   └── main.js            # 入口文件（配置HTTPS请求、全局组件注册）
├── package.json           # 依赖配置
└── vite.config.js         # 构建配置（强制HTTPS，禁用前端SQL拼接）

web-backend/
├── Client/                                # [模块1] 客户端 Web 应用 (端口 8080)
│   ├── pom.xml                            # 依赖：Spring Security, JWT, Web
│   └── src/main/
│       ├── resources/
│       │   └── application.yml            # 配置：JWT密钥, AES密钥, 远程Server地址
│       └── java/com/chd/client/
│           ├── ClientApplication.java     # 启动类
│           ├── config/                    # 配置类
│           │   ├── JwtConfig.java         # JWT 生成配置
│           │   ├── JwtTokenUtil.java      # JWT 解析与校验工具
│           │   ├── SecurityConfig.java    # Spring Security配置 (放行登录, CORS)
│           │   ├── WebMvcConfig.java      # 拦截器注册
│           │   ├── RestTemplateConfig.java# 远程调用工具配置
│           │   ├── PermissionInterceptor.java # 权限拦截器 (解析Token, 校验角色)
│           │   └── RequirePermission.java # 自定义权限注解 (@interface)
│           ├── controller/
│           │   └── RemoteClientController.java # 统一 Web 接口 (登录, 查成绩, 改密码)
│           ├── service/
│           │   └── RemoteClientService.java    # 核心业务逻辑 (加密处理, 调用 Server RPC)
│           ├── dto/                       # 数据传输对象
│           │   ├── OperationDTO.java      # 通用操作对象 (表名, 操作类型, 数据)
│           │   └── AuditLogDTO.java       # 审计日志对象
│           └── utils/
│               ├── Result.java            # 统一返回结果封装
│               ├── AESUtil.java           # AES 加解密工具 (数据加密)
│               └── BCryptUtil.java        # BCrypt 哈希工具 (密码校验)
│
├── Server/                                # [模块2] 服务端 RPC 应用 (端口 8081)
│   ├── pom.xml                            # 依赖：JDBC, MySQL Driver
│   └── src/main/
│       ├── resources/
│       │   └── application.yml            # 配置：MySQL数据源, AES密钥
│       └── java/com/chd/server/
│           ├── ServerApplication.java     # 启动类
│           ├── config/
│           │   └── SecurityConfig.java    # 安全配置 (放行 /rpc/** 供内部调用)
│           ├── controller/
│           │   └── RpcProviderController.java # RPC 接口暴露 (/select, /insert, /update)
│           ├── Interface/                 # 业务接口层
│           │   ├── SelectInterface.java       # 查询接口定义
│           │   ├── ManipulationInterface.java # 增改接口定义
│           │   └── AuditInterface.java        # 审计接口定义
│           ├── impl/                      # 接口实现层 (直接操作 JDBC)
│           │   ├── SelectInterfaceImpl.java       # SQL 查询实现 (含表名校验)
│           │   ├── ManipulationInterfaceImpl.java # SQL 拼接与执行 (INSERT/UPDATE)
│           │   └── AuditInterfaceImpl.java        # 审计日志入库实现
│           ├── dto/                       # 数据传输对象 (需与 Client 端保持一致)
│           │   ├── OperationDTO.java
│           │   └── AuditLogDTO.java
│           └── utils/
│               └── AESUtil.java           # AES 工具 (用于服务端解密/加密数据)
